486  {
487  BLAZE_CONSTRAINT_MUST_BE_VECTOR_WITH_TRANSPOSE_FLAG( VT2, TF );

490

492 
493  const size_t ipos( n & size_t(-2) );
494  BLAZE_INTERNAL_ASSERT( ( n - ( n % 2UL ) ) == ipos, "Invalid end calculation" );
495

497  dv_[i ] = (~rhs)[i ];
